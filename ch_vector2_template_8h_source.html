20  // Compute the perpendicular using the formal determinant,
21  // perp = det{{e0,e1},{x0,x1}} = (x1,-x0)
22  // where e0 = (1,0), e1 = (0,1), and v = (x0,x1).
23  template <typename Real>
24  Vector2T<Real>
25  perp(Vector2T<Real> const& v) {
26  return Vector2T<Real>{ v[1], -v[0] };
27  }

31  Vector2T<Real>
32  unitPerp(Vector2T<Real> const& v, bool robust = false) {
33  Vector2T<Real> unitPerp{ v[1], -v[0] };
34  normalize(unitPerp, robust);

42  dotPerp(Vector2T<Real> const& v0, Vector2T<Real> const& v1) {

46  // Compute a right-handed orthonormal basis for the orthogonal complement
47  // of the input vectors. The function returns the smallest length of the
48  // unnormalized vectors computed during the process. If this value is
49  // nearly zero, it is possible that the inputs are linearly dependent
50  // (within numerical round-off errors). On input, numInputs must be 1 and
51  // v[0] must be initialized. On output, the vectors v[0] and v[1] form an
52  // orthonormal set.
53  template <typename Real>
54  Real
55  computeOrthogonalComplement(int numInputs, Vector2T<Real>* v, bool robust = false) {
56  if (numInputs == 1) {
57  v[1] = -perp(v[0]);
58  return orthonormalize<Real, 2>(2, v, robust);
59  }
60  return static_cast<Real>(0);
61  }

63  // Compute the barycentric coordinates of the point P with respect to the
64  // triangle <V0,V1,V2>, P = b0*V0 + b1*V1 + b2*V2, where b0 + b1 + b2 = 1.
65  // The return value is 'true' iff {V0,V1,V2} is a linearly independent
66  // set. Numerically, this is measured by |det[V0 V1 V2]| <= epsilon. The
67  // values bary[] are valid only when the return value is 'true' but set to
68  // zero when the return value is 'false'.
69  template <typename Real>
70  bool
71  computeBarycentrics(Vector2T<Real> const& p, Vector2T<Real> const& v0,
72  Vector2T<Real> const& v1, Vector2T<Real> const& v2, Real bary[3],
73  Real epsilon = static_cast<Real>(0)) {
74  // Compute the vectors relative to V2 of the triangle.
75  Vector2T<Real> diff[3] = { v0 - v2, v1 - v2, p - v2 };
76  Real det = dotPerp(diff[0], diff[1]);
77  if (det < -epsilon || det > epsilon) {
78  Real invDet = static_cast<Real>(1) / det;
79  bary[0] = dotPerp(diff[2], diff[1]) * invDet;
80  bary[1] = dotPerp(diff[0], diff[2]) * invDet;
81  bary[2] = static_cast<Real>(1) - bary[0] - bary[1];
82  return true;
83  }
84  for (int i = 0; i < 3; ++i) {
85  bary[i] = static_cast<Real>(0);
86  }
87  return false;
88  }

90  // Get intrinsic information about the input array of vectors. The return
91  // value is 'true' iff the inputs are valid (numVectors > 0, v is not
92  // null, and epsilon >= 0), in which case the class members are valid.
93  template <typename Real>
94  class IntrinsicsVector2
95  {
96  public:
97  // The constructor sets the class members based on the input set.
98  IntrinsicsVector2(int numVectors, Vector2T<Real> const* v, Real inEpsilon) :
99  epsilon(inEpsilon),
100  dimension(0),
101  maxRange(static_cast<Real>(0)),
102  origin{ static_cast<Real>(0), static_cast<Real>(0) },
103  extremeCCW(false) {
104  min[0] = static_cast<Real>(0);
105  min[1] = static_cast<Real>(0);
106  direction[0] = { static_cast<Real>(0), static_cast<Real>(0) };
107  direction[1] = { static_cast<Real>(0), static_cast<Real>(0) };
108  extreme[0] = 0;
109  extreme[1] = 0;
110  extreme[2] = 0;
111  if (numVectors > 0 && v && epsilon >= static_cast<Real>(0)) {
112  // Compute the axis-aligned bounding box for the input
113  // vectors. Keep track of the indices into 'vectors' for the
114  // current min and max.
115  int j, indexMin[2], indexMax[2];
116  for (j = 0; j < 2; ++j) {
117  min[j] = v[0][j];
118  max[j] = min[j];
119  indexMin[j] = 0;
120  indexMax[j] = 0;
121  }
122  int i;
123  for (i = 1; i < numVectors; ++i) {
124  for (j = 0; j < 2; ++j) {
125  if (v[i][j] < min[j]) {
126  min[j] = v[i][j];
127  indexMin[j] = i;
128  }
129  else if (v[i][j] > max[j]) {
130  max[j] = v[i][j];
131  indexMax[j] = i;
132  }
133  }
134  }
135  // Determine the maximum range for the bounding box.
136  maxRange = max[0] - min[0];
137  extreme[0] = indexMin[0];
138  extreme[1] = indexMax[0];
139  Real range = max[1] - min[1];
140  if (range > maxRange) {
141  maxRange = range;
142  extreme[0] = indexMin[1];
143  extreme[1] = indexMax[1];
144  }
145  // The origin is either the vector of minimum x0-value or
146  // vector of minimum x1-value.
147  origin = v[extreme[0]];
148  // Test whether the vector set is (nearly) a vector.
149  if (maxRange <= epsilon) {
150  dimension = 0;
151  for (j = 0; j < 2; ++j) {
152  extreme[j + 1] = extreme[0];
153  }
154  return;
155  }
156  // Test whether the vector set is (nearly) a line segment. We
157  // need direction[1] to span the orthogonal complement of
158  // direction[0].
159  direction[0] = v[extreme[1]] - origin;
160  normalize(direction[0], false);
161  direction[1] = -perp(direction[0]);
162  // Compute the maximum distance of the points from the line
163  // origin+t*direction[0].
164  Real maxDistance = static_cast<Real>(0);
165  Real maxSign = static_cast<Real>(0);
166  extreme[2] = extreme[0];
167  for (i = 0; i < numVectors; ++i) {
168  Vector2T<Real> diff = v[i] - origin;
169  Real distance = Dot(direction[1], diff);
170  Real sign = (distance > static_cast<Real>(0) ? static_cast<Real>(1) :
171  (distance < static_cast<Real>(0) ? static_cast<Real>(-1) : static_cast<Real>(0)));
172  distance = Math::absFloat(distance);
173  if (distance > maxDistance) {
174  maxDistance = distance;
175  maxSign = sign;
176  extreme[2] = i;
177  }
178  }
179  if (maxDistance <= epsilon * maxRange) {
180  // The points are (nearly) on the line
181  // origin + t * direction[0].
182  dimension = 1;
183  extreme[2] = extreme[1];
184  return;
185  }
186  dimension = 2;
187  extremeCCW = (maxSign > static_cast<Real>(0));
188  return;
189  }
190  }

192  // A nonnegative tolerance that is used to determine the intrinsic
193  // dimension of the set.
194  Real epsilon;
195 
196  // The intrinsic dimension of the input set, computed based on the
197  // nonnegative tolerance mEpsilon.
198  int dimension;
199 
200  // Axis-aligned bounding box of the input set. The maximum range is
201  // the larger of max[0]-min[0] and max[1]-min[1].
202  Real min[2], max[2];
203  Real maxRange;
204 
205  // Coordinate system. The origin is valid for any dimension d. The
206  // unit-length direction vector is valid only for 0 <= i < d. The
207  // extreme index is relative to the array of input points, and is also
208  // valid only for 0 <= i < d. If d = 0, all points are effectively
209  // the same, but the use of an epsilon may lead to an extreme index
210  // that is not zero. If d = 1, all points effectively lie on a line
211  // segment. If d = 2, the points are not collinear.
212  Vector2T<Real> origin;
213  Vector2T<Real> direction[2];
214 
215  // The indices that define the maximum dimensional extents. The
216  // values extreme[0] and extreme[1] are the indices for the points
217  // that define the largest extent in one of the coordinate axis
218  // directions. If the dimension is 2, then extreme[2] is the index
219  // for the point that generates the largest extent in the direction
220  // perpendicular to the line through the points corresponding to
221  // extreme[0] and extreme[1]. The triangle formed by the points
222  // V[extreme[0]], V[extreme[1]], and V[extreme[2]] is clockwise or
223  // counterclockwise, the condition stored in extremeCCW.
224  int extreme[3];
225  bool extremeCCW;
226  };
