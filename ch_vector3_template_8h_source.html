25  // Compute the cross product using the formal determinant:
26  // cross = det{{e0,e1,e2},{x0,x1,x2},{y0,y1,y2}}
27  // = (x1*y2-x2*y1, x2*y0-x0*y2, x0*y1-x1*y0)
28  // where e0 = (1,0,0), e1 = (0,1,0), e2 = (0,0,1), v0 = (x0,x1,x2), and
29  // v1 = (y0,y1,y2).
30  template <int N, typename Real>
31  VectorT<Real, N> Cross(VectorT<Real, N> const& v0, VectorT<Real, N> const& v1)
32  {
33  static_assert(N == 3 || N == 4, "Dimension must be 3 or 4.");
34 
35  VectorT<Real, N> result;
36  result.MakeZero();
37  result[0] = v0[1] * v1[2] - v0[2] * v1[1];
38  result[1] = v0[2] * v1[0] - v0[0] * v1[2];
39  result[2] = v0[0] * v1[1] - v0[1] * v1[0];
40  return result;
41  }

43  // Compute the normalized cross product.
44  template <int N, typename Real>
45  VectorT<Real, N> UnitCross(VectorT<Real, N> const& v0, VectorT<Real, N> const& v1, bool robust = false)
46  {
47  static_assert(N == 3 || N == 4, "Dimension must be 3 or 4.");
48  VectorT<Real, N> unitCross = Cross(v0, v1);
49  Normalize(unitCross, robust);
50  return unitCross;

53  // Compute Dot((x0,x1,x2),Cross((y0,y1,y2),(z0,z1,z2)), the triple scalar
54  // product of three vectors, where v0 = (x0,x1,x2), v1 = (y0,y1,y2), and
55  // v2 is (z0,z1,z2).
56  template <int N, typename Real>
57  Real DotCross(VectorT<Real, N> const& v0, VectorT<Real, N> const& v1,
58  VectorT<Real, N> const& v2)
59  {
60  static_assert(N == 3 || N == 4, "Dimension must be 3 or 4.");
61  return Dot(v0, Cross(v1, v2));

64  // Compute a right-handed orthonormal basis for the orthogonal complement
65  // of the input vectors. The function returns the smallest length of the
66  // unnormalized vectors computed during the process. If this value is
67  // nearly zero, it is possible that the inputs are linearly dependent
68  // (within numerical round-off errors). On input, numInputs must be 1 or
69  // 2 and v[0] through v[numInputs-1] must be initialized. On output, the
70  // vectors v[0] through v[2] form an orthonormal set.
71  template <typename Real>
72  Real computeOrthogonalComplement(int numInputs, Vector3T<Real>* v, bool robust = false)
73  {
74  if (numInputs == 1)
75  {
76  if (std::fabs(v[0][0]) > std::fabs(v[0][1]))
77  {
78  v[1] = { -v[0][2], (Real)0, +v[0][0] };
79  }
80  else
81  {
82  v[1] = { (Real)0, +v[0][2], -v[0][1] };
83  }
84  numInputs = 2;
85  }
86 
87  if (numInputs == 2)
88  {
89  v[2] = Cross(v[0], v[1]);
90  return orthonormalize<Real, 3>(3, v, robust);
91  }
92 
93  return (Real)0;
94  }

96  // Compute the barycentric coordinates of the point P with respect to the
97  // tetrahedron <V0,V1,V2,V3>, P = b0*V0 + b1*V1 + b2*V2 + b3*V3, where
98  // b0 + b1 + b2 + b3 = 1. The return value is 'true' iff {V0,V1,V2,V3} is
99  // a linearly independent set. Numerically, this is measured by
100  // |det[V0 V1 V2 V3]| <= epsilon. The values bary[] are valid only when
101  // the return value is 'true' but set to zero when the return value is
102  // 'false'.
103  template <typename Real>
104  bool computeBarycentrics(Vector3T<Real> const& p, Vector3T<Real> const& v0,
105  Vector3T<Real> const& v1, Vector3T<Real> const& v2, Vector3T<Real> const& v3,
106  Real bary[4], Real epsilon = (Real)0)
107  {
108  // Compute the vectors relative to V3 of the tetrahedron.
109  Vector3T<Real> diff[4] = { v0 - v3, v1 - v3, v2 - v3, p - v3 };
110 
111  Real det = DotCross(diff[0], diff[1], diff[2]);
112  if (det < -epsilon || det > epsilon)
113  {
114  Real invDet = ((Real)1) / det;
115  bary[0] = DotCross(diff[3], diff[1], diff[2]) * invDet;
116  bary[1] = DotCross(diff[3], diff[2], diff[0]) * invDet;
117  bary[2] = DotCross(diff[3], diff[0], diff[1]) * invDet;
118  bary[3] = (Real)1 - bary[0] - bary[1] - bary[2];
119  return true;
120  }
121 
122  for (int i = 0; i < 4; ++i)
123  {
124  bary[i] = (Real)0;
125  }
126  return false;
127  }

129  // Get intrinsic information about the input array of vectors. The return
130  // value is 'true' iff the inputs are valid (numVectors > 0, v is not
131  // null, and epsilon >= 0), in which case the class members are valid.
132  template <typename Real>
133  class IntrinsicsVector3
134  {
135  public:
136  // The constructor sets the class members based on the input set.
137  IntrinsicsVector3(int numVectors, Vector3T<Real> const* v, Real inEpsilon)
138  :
139  epsilon(inEpsilon),
140  dimension(0),
141  maxRange((Real)0),
142  origin{ (Real)0, (Real)0, (Real)0 },
143  extremeCCW(false)
144  {
145  min[0] = (Real)0;
146  min[1] = (Real)0;
147  min[2] = (Real)0;
148  direction[0] = { (Real)0, (Real)0, (Real)0 };
149  direction[1] = { (Real)0, (Real)0, (Real)0 };
150  direction[2] = { (Real)0, (Real)0, (Real)0 };
151  extreme[0] = 0;
152  extreme[1] = 0;
153  extreme[2] = 0;
154  extreme[3] = 0;
155 
156  if (numVectors > 0 && v && epsilon >= (Real)0)
157  {
158  // Compute the axis-aligned bounding box for the input vectors.
159  // Keep track of the indices into 'vectors' for the current
160  // min and max.
161  int j, indexMin[3], indexMax[3];
162  for (j = 0; j < 3; ++j)
163  {
164  min[j] = v[0][j];
165  max[j] = min[j];
166  indexMin[j] = 0;
167  indexMax[j] = 0;
168  }
169 
170  int i;
171  for (i = 1; i < numVectors; ++i)
172  {
173  for (j = 0; j < 3; ++j)
174  {
175  if (v[i][j] < min[j])
176  {
177  min[j] = v[i][j];
178  indexMin[j] = i;
179  }
180  else if (v[i][j] > max[j])
181  {
182  max[j] = v[i][j];
183  indexMax[j] = i;
184  }
185  }
186  }
187 
188  // Determine the maximum range for the bounding box.
189  maxRange = max[0] - min[0];
190  extreme[0] = indexMin[0];
191  extreme[1] = indexMax[0];
192  Real range = max[1] - min[1];
193  if (range > maxRange)
194  {
195  maxRange = range;
196  extreme[0] = indexMin[1];
197  extreme[1] = indexMax[1];
198  }
199  range = max[2] - min[2];
200  if (range > maxRange)
201  {
202  maxRange = range;
203  extreme[0] = indexMin[2];
204  extreme[1] = indexMax[2];
205  }
206 
207  // The origin is either the VectorT of minimum x0-value, VectorT
208  // of minimum x1-value, or VectorT of minimum x2-value.
209  origin = v[extreme[0]];
210 
211  // Test whether the VectorT set is (nearly) a VectorT.
212  if (maxRange <= epsilon)
213  {
214  dimension = 0;
215  for (j = 0; j < 3; ++j)
216  {
217  extreme[j + 1] = extreme[0];
218  }
219  return;
220  }
221 
222  // Test whether the VectorT set is (nearly) a line segment. We
223  // need {direction[2],direction[3]} to span the orthogonal
224  // complement of direction[0].
225  direction[0] = v[extreme[1]] - origin;
226  Normalize(direction[0], false);
227  if (std::fabs(direction[0][0]) > std::fabs(direction[0][1]))
228  {
229  direction[1][0] = -direction[0][2];
230  direction[1][1] = (Real)0;
231  direction[1][2] = +direction[0][0];
232  }
233  else
234  {
235  direction[1][0] = (Real)0;
236  direction[1][1] = +direction[0][2];
237  direction[1][2] = -direction[0][1];
238  }
239  Normalize(direction[1], false);
240  direction[2] = Cross(direction[0], direction[1]);
241 
242  // Compute the maximum distance of the points from the line
243  // origin + t * direction[0].
244  Real maxDistance = (Real)0;
245  Real distance, dot;
246  extreme[2] = extreme[0];
247  for (i = 0; i < numVectors; ++i)
248  {
249  Vector3T<Real> diff = v[i] - origin;
250  dot = Dot(direction[0], diff);
251  Vector3T<Real> proj = diff - dot * direction[0];
252  distance = Length(proj, false);
253  if (distance > maxDistance)
254  {
255  maxDistance = distance;
256  extreme[2] = i;
257  }
258  }
259 
260  if (maxDistance <= epsilon * maxRange)
261  {
262  // The points are (nearly) on the line
263  // origin + t * direction[0].
264  dimension = 1;
265  extreme[2] = extreme[1];
266  extreme[3] = extreme[1];
267  return;
268  }
269 
270  // Test whether the VectorT set is (nearly) a planar polygon.
271  // The point v[extreme[2]] is farthest from the line:
272  // origin + t * direction[0]. The VectorT
273  // v[extreme[2]] - origin is not necessarily perpendicular to
274  // direction[0], so project out the direction[0] component so
275  // that the result is perpendicular to direction[0].
276  direction[1] = v[extreme[2]] - origin;
277  dot = Dot(direction[0], direction[1]);
278  direction[1] -= dot * direction[0];
279  Normalize(direction[1], false);
280 
281  // We need direction[2] to span the orthogonal complement of
282  // {direction[0],direction[1]}.
283  direction[2] = Cross(direction[0], direction[1]);
284 
285  // Compute the maximum distance of the points from the plane
286  // origin+t0 * direction[0] + t1 * direction[1].
287  maxDistance = (Real)0;
288  Real maxSign = (Real)0;
289  extreme[3] = extreme[0];
290  for (i = 0; i < numVectors; ++i)
291  {
292  Vector3T<Real> diff = v[i] - origin;
293  distance = Dot(direction[2], diff);
294  Real sign = (distance > (Real)0 ? (Real)1 :
295  (distance < (Real)0 ? (Real)-1 : (Real)0));
296  distance = std::fabs(distance);
297  if (distance > maxDistance)
298  {
299  maxDistance = distance;
300  maxSign = sign;
301  extreme[3] = i;
302  }
303  }
304 
305  if (maxDistance <= epsilon * maxRange)
306  {
307  // The points are (nearly) on the plane
308  // origin + t0 * direction[0] + t1 * direction[1].
309  dimension = 2;
310  extreme[3] = extreme[2];
311  return;
312  }
313 
314  dimension = 3;
315  extremeCCW = (maxSign > (Real)0);
316  return;
317  }
318  }

320  // A nonnegative tolerance that is used to determine the intrinsic
321  // dimension of the set.
322  Real epsilon;
323 
324  // The intrinsic dimension of the input set, computed based on the
325  // nonnegative tolerance mEpsilon.
326  int dimension;
327 
328  // Axis-aligned bounding box of the input set. The maximum range is
329  // the larger of max[0]-min[0], max[1]-min[1], and max[2]-min[2].
330  Real min[3], max[3];
331  Real maxRange;
332 
333  // Coordinate system. The origin is valid for any dimension d. The
334  // unit-length direction VectorT is valid only for 0 <= i < d. The
335  // extreme index is relative to the array of input points, and is also
336  // valid only for 0 <= i < d. If d = 0, all points are effectively
337  // the same, but the use of an epsilon may lead to an extreme index
338  // that is not zero. If d = 1, all points effectively lie on a line
339  // segment. If d = 2, all points effectively line on a plane. If
340  // d = 3, the points are not coplanar.
341  Vector3T<Real> origin;
342  Vector3T<Real> direction[3];
343 
344  // The indices that define the maximum dimensional extents. The
345  // values extreme[0] and extreme[1] are the indices for the points
346  // that define the largest extent in one of the coordinate axis
347  // directions. If the dimension is 2, then extreme[2] is the index
348  // for the point that generates the largest extent in the direction
349  // perpendicular to the line through the points corresponding to
350  // extreme[0] and extreme[1]. Furthermore, if the dimension is 3,
351  // then extreme[3] is the index for the point that generates the
352  // largest extent in the direction perpendicular to the triangle
353  // defined by the other extreme points. The tetrahedron formed by the
354  // points V[extreme[0]], V[extreme[1]], V[extreme[2]], and
355  // V[extreme[3]] is clockwise or counterclockwise, the condition
356  // stored in extremeCCW.
357  int extreme[4];
358  bool extremeCCW;
359  };
